my_plot, = plt.plot(St, c, 'bo')

def run(i):
    d1 = (np.log(St/K) + (r + 0.5*sigma**2)*np.sqrt(T+0.01*i-t))/(sigma*(np.sqrt(T+0.01*i-t)))
    d2 = d1 - sigma*(np.sqrt(T+0.01*i-t))
    c = St*norm.cdf(d1) - K*norm.cdf(d2)*np.exp(1)**(-r*(T+0.01*i-t))
    delta = norm.cdf(d1)
    my_text.set_text(r'$r = 0.05, \sigma = 0.5, \tau =$ %.2f' % (T+0.01*i-t))
    my_plot.set_ydata(c)
    return my_plot,

def init():
    return my_plot,

ani = animation.FuncAnimation(fig, run, init_func=init, frames=100, interval=25, blit=True)
