-- CREATE TABLE my_table (my_column_1 data_type_1, my_column_2 data_type_1, my_column_3 data_type_2)
CREATE TABLE student (
    first_name VARCHAR(30) NOT NULL,    -- "first_name" is the name of column; "VARCHAR" is its data type; "NOT NULL" makes sure the data is entered(be in the database)
    last_name VARCHAR(30) NOT NULL,
    email VARCHAR(60) NULL, -- not necessarily be filled in
    state CHAR(2) NOT NULL DEFAULT "MA", -- by default the value is "MA"
    zip MEDIUMINT UNSIGNED NOT NULL,
    birth_date DATE NOT NULL,
    sex ENUM('M', 'F') NOT NULL,
    date_entered TIMESTAMP, -- gives data and time
    lunch_cost FLOAT NULL,
    student_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY);   -- with "AUTO_INCREMENT", no need to input; set "student_id" as the PRIMARY KEY

/* 1. Primary keys is unique. 
   2. It cannot be NULL.
   3. Original value cannot be changed.
   4. ? It is probably best to auto-increment the value of the key.

   Tables:
   1. Every table should just focus on one thing.
   2. Then decide the things you need to describe this one thing.
   3. If one description needs multiple inputs, pull it out.(So, we call them atomic tables.)
   4. Don't have multiple columns with same sort of information.(eg, travel_history_1, travel_histroy_2)
   5. Don't include multiple values in one cell.(eg. jobs: Goldman, Blackstone)
   6. Normalized tables.(Database normalization is the process of organizing the columns and tables of a relational database to minimize data redundancy.)
*/

-- Datatypes:
-- numeric types:
TINYINT:    127 ~ -128
SMALLINT:   32768 ~ -36767
MEDIUMINT:  8388608 ~ -8388608
INT:        2^31 ~ (-2^31-1)
BIGINT:     2^31 ~ (-2^63-1)

FLOAT:      Decimal spaces: 1.1E38 ~ -1.1E38
DOUBLE:     Decimal spaces: 1.7E308 ~ -1.7E308

-- string types:
CHAR:       fixed length character string
VARCHAR:    A character string with a variable length
BLOB:       Binary Large Object (BLOB): can contain 2^16 bytes of data

ENUM:       limited number of total values.(eg. "male" or "femail")
SET:        list of values

-- time types:
DATE:       YYYY-MM-DD
TIME:       HH:MM:SS
DATETIME:   YYYY-MM-DD HH:MM:SS
TIMESTAMP:  YYYYMMDDHHMMSS
YEAR:       YYYY

CREATE TABLE test(
date DATE NOT NULL,
type ENUM('T', 'Q') NOT NULL,
class_id INT UNSIGNED NOT NULL, -- "class_id" is the foreign key
test_id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY);

/* Foreign Key:
   1. Used to make references to the Primary Key of another table.
   2. The foreign Key can have a different name from the Primary Key name.
   3. The value of a Foreign Key can have the value of NULL.
   4. A Foreign Key does not have to be unique.
*/

CREATE TABLE score(

event_id INT UNSIGNED NOT NULL,

-- INNER JOIN: Get all the data from both tables and join them together.
SELECT first_name
FROM students
INNER JOIN scores
ON students.student_id=scores.student_id
WHERE scores.score<=15
GROUP BY scores.test_id;

-- LEFT JOIN: show everything of the table on the left even the right one has no corresponding value
SELECT students.student_id,
COUNT(absences.date) AS Absences
FROM students LEFT JOIN absences
ON students.student_id=absences.student_id  -- The two things that both tables have in common
GROUP BY students.student_id;
